def machine_lpar_list

    #column(1).width = 40

    #column(8).width = 50

  end

def machine_lpar_list_row
  [["LPAR", "Machine"]] +
    @machine_lpars.map do |machine_lpar|
    [ machine_lpar.lpar.name, machine_lpar.machine.name ]
  end
end
