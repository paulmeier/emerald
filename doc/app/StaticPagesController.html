def cpureport
  @machines = Machine.includes(:lpars).includes(:machineConfig).all
  
  @peaks = Cpu.new
  @mAvg = Cpu.new
  @mPeak = Cpu.new   
  
   #Build each machine's LPAR's peaks and average. mAvg builds the machines total box average.
   @machines.each_with_index do |m,i|
     @peaks[i] = Cpu.find(:all, :group => "smfid", :select => "smfid, max(mips) as max, AVG(mips) as average", :conditions => ['smfid in (?)', m.lpars.map(&:smfid)])
     @mPeak[i] = Cpu.find(:all, :group => "date(datetime),time(datetime)", :select => "datetime ,sum(mips) as peak", :order => "peak DESC", :conditions => ['smfid in (?)', m.lpars.map(&:smfid)])
     @mAvg[i] = Cpu.find(:all, :select => "AVG(mips) as average", :conditions => ['smfid in (?)', m.lpars.map(&:smfid)])     
   end
end
