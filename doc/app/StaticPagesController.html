def ziipreport    
  #Get the locations machines.
  @machines = Location.find(params[:location]).machines
  
  #Make sure the machine has a ZIIP. If not, dont even bother querying for information.
  @machines.delete_if{ |x| x.machineConfig.ziip <= 0 }
end
